CREATE TABLE project(
   idproject SERIAL,
   nameproject VARCHAR(50),
   dateproject VARCHAR(50),
   descriptionproject TEXT,
   PRIMARY KEY(idproject)
);
CREATE TABLE roles(
   idrole SERIAL,
   namerole VARCHAR(50),
   PRIMARY KEY(idrole)
);
CREATE TABLE techs(
   idtech SERIAL,
   nametech VARCHAR(50),
   PRIMARY KEY(idtech)
);
CREATE TABLE roleproject(
   idproject INTEGER,
   idrole INTEGER,
   PRIMARY KEY(idproject, idrole),
   FOREIGN KEY(idproject) REFERENCES project(idproject),
   FOREIGN KEY(idrole) REFERENCES roles(idrole)
);
CREATE TABLE techproject(
   idproject INTEGER,
   idtech INTEGER,
   PRIMARY KEY(idproject, idtech),
   FOREIGN KEY(idproject) REFERENCES project(idproject),
   FOREIGN KEY(idtech) REFERENCES techs(idtech)
);
